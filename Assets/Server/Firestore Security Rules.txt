rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {  
    match /global/{documentId} {
    	allow read: if true;
      allow write: if false;
    }
    
    match /users/{documentId} {
    	allow read: if request.auth.token.name == existingData().myId;
      allow create: if
      	request.auth.token.name == incomingData().myId &&
        isValidString("password", 12, 12);
      allow update: if false;
      allow delete: if false;
    }

    match /messageUpdates/{documentId} {
    	allow read: if
      	request.auth.token.name == existingData().myId ||
        request.auth.token.name == existingData().buddyId;
      allow create: if request.auth.token.name == incomingData().myId;
      allow update: if
      	request.auth.token.name == incomingData().myId &&
        isNotUpdating("id") &&
        isNotUpdating("myId") &&
        isNotUpdating("buddyId");
      allow delete: if false;
    }

    match /messages/{documentId} {
    	allow read: if false;
      allow write: if false;
    }
    
    function isNotUpdating(field) {
			return
    		!(field in incomingData()) ||
  			existingData()[field] == incomingData()[field];
		}
    
    function existingData() {
    	return resource.data;
    }
    
    function incomingData() {
    	return request.resource.data;
    }
  	
  	function isValidString(field, minLength, maxLength) {
  		return
    		incomingData()[field] is string &&
    		incomingData()[field].size() >= minLength &&
    	  incomingData()[field].size() <= maxLength;
  	}
  }
}
